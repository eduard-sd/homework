mkdir hello - создание папки hello
cd hello - открываем директорию hello (копирование строки из проводника ctrl+c, shift+insert - вставить в гит баш, поменять слэши на обратные)
touch hello.html - создает файл hello.html

git init - Создайте новый репозиторий


git add hello.html -  добавить в отслеживание файл хелло.  Если вы хотите добавить изменения в репозиторий, используйте команду. 
git add . -  «.» это самый краткий и удобный путь для добавления всех изменений в файлы текущего каталога и его подкаталоги

git commit -m "First Commit" - добавить комментарий к файлу хелло/ Если вы опустите метку -m из командной строки, git перенесет вас в редактор (выход - ESC, ввести :wq и нажать Enter)


git status - чтобы проверить текущее состояние репозитория.


git сheckout - для отмены изменений.
git reset - Если вы решили, что не хотите коммитить изменения, команда состояния напомнит вам о том, что с помощью команды git reset можно снять индексацию этих изменений. ОТмена git add


git log - Получение списка произведенных изменений
git log --pretty=oneline  -  однострочный формат log

git log --pretty=oneline --max-count=2
git log --pretty=oneline --since='5 minutes ago'
git log --pretty=oneline --until='5 minutes ago'
git log --pretty=oneline --author=<your name>
git log --pretty=oneline --all
git log --pretty=format:"%h %ad | %s%d [%an]" --graph --date=short   - мне подходит следующий формат лога

--pretty="..."  —  определяет формат вывода.
%h  —  укороченный хэш коммита
%d  —  дополнения коммита («головы» веток или теги)
%ad  —  дата коммита
%s  —  комментарий
%an  —  имя автора
--graph  —  отображает дерево коммитов в виде ASCII-графика
--date=short  —  сохраняет формат даты коротким и симпатичным
